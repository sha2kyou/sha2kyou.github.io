# SideScript 1.7

IF title CONTAINS_ANY ["keyword1", "keyword2", "keyword3"]
  AND title NOT CONTAINS_ANY ["keyword4", "keyword5"]
  AND origin NOT CONTAINS "BBC"
THEN HIGHLIGHT ["keyword1", "keyword2", "keyword3"] WITH "#20fefe"
SET color = "#996633", symbol = "star.fill";

IF title CONTAINS_ANY ["morning", "daily", "news"]
  OR origin CONTAINS_ANY ["Notification"]
  OR source CONTAINS_ANY ["reminders", "calendar"]
THEN SET PINNED;

IF anywhere CONTAINS_ANY ["country", "region"]
  AND source CONTAINS "RSS"
THEN SET symbol = "star.fill";

IF title CONTAINS_ANY ["person1", "person2", "company1"]
THEN SET symbol = "theatermasks.fill";
